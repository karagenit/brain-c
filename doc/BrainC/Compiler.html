def add(value)
  value = value.to_i
  if value > 0
    value.times { @out << '+' }
  else
    value.abs.times { @out << '-' }
  end
end

def compile(inst)
  raise ArgumentError if inst.nil?
  fields = inst.split(' ')

  case fields.first
  when 'int', 'char'
    vars[fields[1]] = alloc_var
  when /putchar\((.*)\)/
    puts "Printing var #{Regexp.last_match(1)}"
  else
    case fields[1]
    when '='
      set_var(fields[0], fields[2])
    end
  end
end

def jump(index)
  index = index.to_i
  diff = index - @index
  if diff > 0
    diff.times { @out << '>' }
  else
    puts "targ: #{index} ind: #{@index} run: #{diff.abs}"
    diff.abs.times { @out << '<' }
  end
end

def preprocess
  @src = @src.delete("\n")
  @src = @src[/int main\(\) \{(.*)\}/, 1]
  @ins = @src.split(';')
  @ins.map! { |inst| inst.chomp.strip }
end
